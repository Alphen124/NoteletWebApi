--คำนวณ RentDuration + TotalPrice
CREATE OR REPLACE FUNCTION fn_rentlist_calc()
RETURNS TRIGGER AS $$
DECLARE
    v_start_date DATE;
    v_end_date   DATE;
    v_price      NUMERIC;
BEGIN
    SELECT StartDate, EndDate
    INTO v_start_date, v_end_date
    FROM Schedule
    WHERE ScheduleNo = NEW.ScheduleNo;

    SELECT RentPrice
    INTO v_price
    FROM Device
    WHERE DeviceNo = NEW.DeviceNo;

    NEW.RentDuration :=
        COALESCE(v_end_date, v_start_date, CURRENT_DATE)
      - COALESCE(v_start_date, v_end_date, CURRENT_DATE);

    NEW.TotalPrice := COALESCE(NEW.RentDuration, 0) * COALESCE(v_price, 0);

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_rentlist_calc
BEFORE INSERT OR UPDATE ON RentList
FOR EACH ROW
EXECUTE FUNCTION fn_rentlist_calc();
--Trigger อัปเดต Rating ของ Renter
CREATE OR REPLACE FUNCTION fn_update_renter_rating()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE Renter r
    SET Rating = (
        SELECT COALESCE(AVG(Rating), 0)
        FROM RentBill
        WHERE RenterNo = NEW.RenterNo
    )
    WHERE r.RenterNo = NEW.RenterNo;

    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_update_renter_rating
AFTER INSERT OR UPDATE OF Rating ON RentBill
FOR EACH ROW
EXECUTE FUNCTION fn_update_renter_rating();

--Trigger อัปเดต Rating ของ Device
CREATE OR REPLACE FUNCTION fn_update_device_rating()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE Device d
    SET Rating = (
        SELECT COALESCE(AVG(rb.Rating), 0)
        FROM RentBill rb
        JOIN RentList rl ON rb.RentingNo = rl.RentingNo
        WHERE rl.DeviceNo = d.DeviceNo
    )
    WHERE d.DeviceNo IN (
        SELECT DeviceNo
        FROM RentList
        WHERE RentingNo = NEW.RentingNo
    );

    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_update_device_rating
AFTER INSERT OR UPDATE OF Rating ON RentBill
FOR EACH ROW
EXECUTE FUNCTION fn_update_device_rating();
--Trigger อัปเดต Rating ของ Owner
CREATE OR REPLACE FUNCTION fn_update_owner_rating()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE Owner o
    SET Rating = (
        SELECT COALESCE(AVG(rb.Rating), 0)
        FROM RentBill rb
        JOIN RentList rl ON rb.RentingNo = rl.RentingNo
        JOIN DeviceOwner do2 ON rl.DeviceNo = do2.DeviceNo
        WHERE do2.OwnerNo = o.OwnerNo
    )
    WHERE o.OwnerNo IN (
        SELECT do2.OwnerNo
        FROM RentList rl
        JOIN DeviceOwner do2 ON rl.DeviceNo = do2.DeviceNo
        WHERE rl.RentingNo = NEW.RentingNo
    );

    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_update_owner_rating
AFTER INSERT OR UPDATE OF Rating ON RentBill
FOR EACH ROW
EXECUTE FUNCTION fn_update_owner_rating();


--Procedure insert_rentlist
CREATE OR REPLACE FUNCTION insert_rentlist(
    p_rentlistno  INTEGER,
    p_deviceno    INTEGER,
    p_scheduleno  INTEGER,
    p_rentingno   INTEGER
)
RETURNS VOID AS $$
DECLARE
    v_next_seq INTEGER;
BEGIN
    SELECT COALESCE(MAX(RentListSeq), 0) + 1
    INTO v_next_seq
    FROM RentList
    WHERE RentListNo = p_rentlistno;

    INSERT INTO RentList (
        RentListNo, RentListSeq, DeviceNo, ScheduleNo, RentingNo
    ) VALUES (
        p_rentlistno, v_next_seq, p_deviceno, p_scheduleno, p_rentingno
    );
END;
$$ LANGUAGE plpgsql;
